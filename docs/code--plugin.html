      \AKlump\CheckPages\Event::ASSERT_CREATED => [

      ],
    ];
  }

  /**
   * {@inheritdoc}
   */

    $this->runner = $event->getTest()->getRunner();
    $event->getAssert()->setToStringOverride([$this, 'onAssertToString']);

   */

    return $stringified;
  }

}
